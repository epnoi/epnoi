012package gate.stanford;
013
014import java.io.Serializable;
015
016/**
017 * Simple data structure representing a single dependency relation.  The "target"
018 * is the Annotation ID of the dependent; the "type" is the dependency 
019 * tag (<a href="http://nlp.stanford.edu/software/parser-faq.shtml#c">the
020 * Stanford Parser documentation</a> contains links to the tagset</a>; for example,
021 * nsubj = "nominal subject", dobj = "direct object).
022 */
023public class DependencyRelation implements Serializable {
024
025  private static final long serialVersionUID = -7842607116149222052L;
026
027  /**
028   * The type of the dependency relation (det, amod, etc.).
029   */
030  private String type;
031  
032  /**
033   * The ID of the token that is the target of this relation.
034   */
035  private Integer targetId;
036  
037  public DependencyRelation(String type, Integer targetId) {
038    this.type = type;
039    this.targetId = targetId;
040  }
041
042  /**
043   * Return the dependency tag (type).
044   * @return the dependency tag
045   */
046  public String getType() {
047    return type;
048  }
049
050  /**
051   * Set the dependency tag.
052   * @param type dependency tag
053   */
054  public void setType(String type) {
055    this.type = type;
056  }
057
058  /**
059   * Return the GATE Annotation ID of the dependent.
060   * @return the Annotation ID
061   */
062  public Integer getTargetId() {
063    return targetId;
064  }
065
066  /**
067   * Set the Annotation ID of the dependent.
068   * @param targetId the Annotation ID
069   */
070  public void setTargetId(Integer targetId) {
071    this.targetId = targetId;
072  }
073  
074  /**
075   * Format the data structure for display.
076   * For example, if type is "dobj" and the dependent has Annotation ID 37,
077   * return the String "dobj(37)". 
078   */
079  public String toString() {
080    return type + "(" + targetId + ")";
081  }
082}
