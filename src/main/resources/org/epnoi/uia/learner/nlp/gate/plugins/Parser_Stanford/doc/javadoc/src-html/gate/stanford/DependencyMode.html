012package gate.stanford;
013
014import java.util.Collection;
015
016import edu.stanford.nlp.trees.GrammaticalStructure;
017import edu.stanford.nlp.trees.TypedDependency;
018
019public enum DependencyMode {
020  Typed,
021  AllTyped,
022  TypedCollapsed,
023  TypedCCprocessed;
024  
025  
026  protected static Collection<TypedDependency> getDependencies(GrammaticalStructure gs,
027      DependencyMode mode, boolean includeExtras) {
028    Collection<TypedDependency> result = null;
029    
030    if (mode.equals(Typed)) {
031      result = gs.typedDependencies(includeExtras);
032    }
033    else if (mode.equals(AllTyped)) {
034      result = gs.allTypedDependencies();
035    }
036    else if (mode.equals(TypedCollapsed)) {
037      result = gs.typedDependenciesCollapsed(includeExtras);
038    }
039    else if (mode.equals(TypedCCprocessed)) {
040      result = gs.typedDependenciesCCprocessed(includeExtras);
041    }
042    
043    return result;
044  }
045
046}
